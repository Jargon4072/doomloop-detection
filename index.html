troubleshooter/agent.py
Anthropic SDK
import anthropic, json
from .payload import TroubleshooterPayload
from .output  import TroubleshooterOutput
from .prompts import DIAGNOSIS_SYSTEM_PROMPT

client = anthropic.Anthropic()

def run_troubleshooter(
    payload: TroubleshooterPayload
) -> TroubleshooterOutput:
    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=1000,
        system=DIAGNOSIS_SYSTEM_PROMPT,
        messages=[{
            "role": "user",
            "content": json.dumps(
                payload.__dict__, indent=2
            )
        }]
    )
    raw = response.content[0].text
    return json.loads(raw)
